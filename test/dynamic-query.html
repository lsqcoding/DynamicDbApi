查询示例

                                

                                

                                    

                                        
基本查询

                                        
{
  "table": { "name": "users" },
  "fields": ["id", "name", "email"],
  "page": 1,
  "pageSize": 10
}

                                    


                                    

                                        
条件查询

                                        
{
  "table": { "name": "products" },
  "conditions": [
    { "field": "price", "operator": "__gte", "value": 100 },
    { "field": "category", "operator": "__eq", "value": "electronics" }
  ]
}

                                    


                                    

                                        
排序和分页

                                        
{
  "table": { "name": "orders" },
  "sort": [
    { "field": "order_date", "direction": "desc" },
    { "field": "total_amount", "direction": "asc" }
  ],
  "page": 2,
  "pageSize": 20,
  "includeTotal": true
}

                                    


                                    

                                        
模糊查询

                                        
{
  "table": { "name": "users" },
  "conditions": [
    { "field": "name", "operator": "__like", "value": "%John%" }
  ]
}

                                    


                                    

                                        
IN查询

                                        
{
  "table": { "name": "products" },
  "conditions": [
    { "field": "status", "operator": "__in", "value": ["active", "pending"] }
  ]
}

    "email": "test@example.com",

    "address": "测试地址"

}

                                    


                                    

                                        
带自定义返回值的插入操作

                                        
{
  "table": { "name": "users" },
  "operation": "insert",
  "data": {
    "name": "自定义返回值测试用户",
    "email": "custom-return@example.com",
    "phone": "13800138001",
    "address": "自定义返回值测试地址"
  },
  "returnType": "SingleObject",
  "returnFields": ["id", "name", "email"]
}

                                    


                                    

                                        
带自定义返回值的批量插入操作

      "name": "批量用户1",

基本CTE查询

                                        
{
  "operation": "cte",
  "cteQuery": {
    "definitions": [
      {
        "name": "CTE_UserOrders",
        "query": "SELECT u.id AS UserId, u.name AS UserName, COUNT(o.id) AS OrderCount FROM users u JOIN orders o ON u.id = o.user_id GROUP BY u.id, u.name"
      }
    ]
  },
  "table": { "name": "CTE_UserOrders" },
  "fields": ["UserId", "UserName", "OrderCount"],
  "conditions": [
    { "field": "OrderCount", "operator": "__gte", "value": 5 }
  ],
  "sort": [
    { "field": "OrderCount", "direction": "desc" }
  ]
}

                                    


                                    

                                        
递归CTE查询

                                        
{
  "operation": "cte",
  "cteQuery": {
    "definitions": [
      {
        "name": "CTE_RecursiveCategory",
        "query": "WITH RECURSIVE CTE_RecursiveCategory (id, name, parent_id, level) AS (SELECT id, name, parent_id, 0 FROM categories WHERE parent_id IS NULL UNION ALL SELECT c.id, c.name, c.parent_id, rc.level + 1 FROM categories c INNER JOIN CTE_RecursiveCategory rc ON c.parent_id = rc.id) SELECT * FROM CTE_RecursiveCategory",
        "isRecursive": true
      }
    ]
  },
  "table": { "name": "CTE_RecursiveCategory" },
  "fields": ["id", "name", "parent_id", "level"],
  "sort": [
    { "field": "level", "direction": "asc" },
    { "field": "name", "direction": "asc" }
  ]
}
